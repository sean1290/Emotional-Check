2. 아래 코드 붙여넣고 저장

function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var sheetName = data.session === 'morning' ? '입실' : '퇴실'; var sheet = ss.getSheetByName(sheetName) || ss.insertSheet(sheetName); if (sheet.getLastRow() === 0) sheet.appendRow(Object.keys(data)); sheet.appendRow(Object.values(data)); return ContentService.createTextOutput( JSON.stringify({result:'ok'}) ).setMimeType(ContentService.MimeType.JSON); }
